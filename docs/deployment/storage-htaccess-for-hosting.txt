<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteBase /storage/
    
    # Redirect ke storage/app/public dengan path absolut
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteCond %{REQUEST_FILENAME} !-d
    RewriteRule ^(.*)$ /var/www/finwa_web_id_usr/data/www/finwa.web.id/storage/app/public/$1 [L]
</IfModule>

# Prevent directory listing
Options -Indexes

# Allow access to all file types
<FilesMatch ".*">
    Require all granted
</FilesMatch>
